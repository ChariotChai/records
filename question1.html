import java.util.*;

public class BooleanExpressionParser {

    static class TreeNode {
        String value;
        TreeNode left;
        TreeNode right;

        TreeNode(String value) {
            this.value = value;
            this.left = null;
            this.right = null;
        }
    }

    // 解析布尔表达式为二叉树
    public static TreeNode parse(String expression) {
        // 清理输入，去除空格
        expression = expression.replaceAll("\\s+", "");

        // 使用递归下降解析
        return parseOr(expression, new int[] {0});
    }

    // 解析 OR 表达式
    private static TreeNode parseOr(String expression, int[] index) {
        TreeNode left = parseAnd(expression, index);

        while (index[0] < expression.length() && expression.charAt(index[0]) == '|') {
            index[0]++; // 跳过 '|'
            TreeNode node = new TreeNode("|");
            node.left = left;
            node.right = parseAnd(expression, index);
            left = node; // 更新左子树
        }

        return left;
    }

    // 解析 AND 表达式
    private static TreeNode parseAnd(String expression, int[] index) {
        TreeNode left = parseNot(expression, index);

        while (index[0] < expression.length() && expression.charAt(index[0]) == '&') {
            index[0]++; // 跳过 '&'
            TreeNode node = new TreeNode("&");
            node.left = left;
            node.right = parseNot(expression, index);
            left = node; // 更新左子树
        }

        return left;
    }

    // 解析 NOT 表达式
    private static TreeNode parseNot(String expression, int[] index) {
        if (index[0] < expression.length() && expression.charAt(index[0]) == '!') {
            index[0]++; // 跳过 '!'
            TreeNode node = new TreeNode("!");
            node.right = parseFactor(expression, index); // 取非操作只有右子树
            return node;
        }
        return parseFactor(expression, index);
    }

    // 解析基本的因子：布尔变量或括号
    private static TreeNode parseFactor(String expression, int[] index) {
        if (index[0] < expression.length() && expression.charAt(index[0]) == '(') {
            index[0]++; // 跳过 '('
            TreeNode node = parseOr(expression, index); // 递归解析括号中的表达式
            index[0]++; // 跳过 ')'
            return node;
        }

        return parseVariable(expression, index);
    }

    // 解析布尔变量（例如 x1, x20）
    private static TreeNode parseVariable(String expression, int[] index) {
        int start = index[0];
        while (index[0] < expression.length() && (Character.isLetterOrDigit(expression.charAt(index[0])) || expression.charAt(index[0]) == '_')) {
            index[0]++;
        }
        String variable = expression.substring(start, index[0]);
        return new TreeNode(variable);
    }

    // 打印二叉树，用于测试
    public static void printTree(TreeNode node, String indent) {
        if (node != null) {
            System.out.println(indent + node.value);
            printTree(node.left, indent + "  ");
            printTree(node.right, indent + "  ");
        }
    }

    // 测试
    public static void main(String[] args) {
        String expression = "x1 & (x2 | ! x3)";
        TreeNode root = parse(expression);
        printTree(root, "");
    }
}

  3. parseAnd：解析 AND (&) 操作。
  4. parseNot：解析 NOT (!) 操作，优先级最高。
  5. parseFactor：解析一个布尔因子，它可以是布尔变量（如 x1）或者括号内的表达式。
  6. parseVariable：处理布尔变量的解析，如 x1, x20 等。

import java.util.*;

public class BooleanExpressionParser {

    static class TreeNode {
        String value;
        TreeNode left;
        TreeNode right;

        TreeNode(String value) {
            this.value = value;
            this.left = null;
            this.right = null;
        }
    }

    // 解析布尔表达式为二叉树
    public static TreeNode parse(String expression) {
        expression = expression.replaceAll("\\s+", "");  // 去除空格
        return parseExpression(expression);
    }

    // 解析布尔表达式
    private static TreeNode parseExpression(String expression) {
        Stack<TreeNode> operands = new Stack<>();
        Stack<Character> operators = new Stack<>();
        int index = 0;

        while (index < expression.length()) {
            char ch = expression.charAt(index);

            if (ch == '(') {
                // 递归解析括号内的表达式
                int end = findClosingParenthesis(expression, index);
                operands.push(parseExpression(expression.substring(index + 1, end)));
                index = end + 1; // 更新索引
            } else if (ch == '!') {
                // 取非操作，立即应用
                index++;
                operands.push(new TreeNode("!"));
            } else if (ch == '&' || ch == '|') {
                // 操作符处理
                operators.push(ch);
                index++;
            } else if (ch == 'x') {
                // 变量处理，如 x1, x20
                int start = index;
                while (index < expression.length() && (Character.isDigit(expression.charAt(index)) || expression.charAt(index) == '_')) {
                    index++;
                }
                String variable = expression.substring(start, index);
                operands.push(new TreeNode(variable));
            } else {
                index++; // 忽略不需要的字符
            }

            // 处理运算符优先级
            while (!operators.isEmpty() && (operators.peek() == '!' || operators.peek() == '&' || operators.peek() == '|')) {
                char op = operators.peek();
                if (op == '!') {
                    TreeNode operand = operands.pop();
                    TreeNode node = new TreeNode("!");
                    node.right = operand;
                    operands.push(node);
                    operators.pop();
                } else if (op == '&' || op == '|') {
                    TreeNode right = operands.pop();
                    TreeNode left = operands.pop();
                    TreeNode node = new TreeNode(String.valueOf(op));
                    node.left = left;
                    node.right = right;
                    operands.push(node);
                    operators.pop();
                }
            }
        }
        return operands.isEmpty() ? null : operands.pop();
    }

    // 找到匹配的右括号
    private static int findClosingParenthesis(String expression, int index) {
        int count = 1;
        int i = index + 1;
        while (i < expression.length() && count > 0) {
            if (expression.charAt(i) == '(') {
                count++;
            } else if (expression.charAt(i) == ')') {
                count--;
            }
            i++;
        }
        return i - 1; // 返回右括号的索引
    }

    // 打印二叉树，用于测试
    public static void printTree(TreeNode node, String indent) {
        if (node != null) {
            System.out.println(indent + node.value);
            printTree(node.left, indent + "  ");
            printTree(node.right, indent + "  ");
        }
    }

    // 测试
    public static void main(String[] args) {
        String expression = "x1 & (x2 | ! x3)";
        TreeNode root = parse(expression);
        printTree(root, "");
    }
}

关键点：

  1. 括号处理：当遇到左括号 ( 时，我们递归解析括号内的子表达式，直到遇到右括号 ) 为止。这是通过 findClosingParenthesis 方法来找到匹配的右括号实现的。
  2. 运算符优先级：在解析运算符时，我们使用栈来保存操作符。对于每个操作符，首先检查栈顶的操作符优先级，按照优先级（! > & > |）构建二叉树节点。
  3. 操作数和运算符：对于变量（如 x1），我们直接从字符串中提取并构建二叉树节点；对于运算符（如 &, |），我们在栈中存储操作符，并在遇到运算符时进行二叉树节点的合并。

代码实现

java
import java.util.*;

public class BooleanExpressionParser {

    static class TreeNode {
        String value;
        TreeNode left;
        TreeNode right;

        TreeNode(String value) {
            this.value = value;
            this.left = null;
            this.right = null;
        }
    }

    // 解析切分好的token数组，返回二叉树
    public static TreeNode parse(List<String> tokens) {
        Stack<TreeNode> operands = new Stack<>();  // 操作数栈
        Stack<String> operators = new Stack<>();   // 运算符栈

        int index = 0;
        while (index < tokens.size()) {
            String token = tokens.get(index);

            if (token.equals("(")) {
                // 递归解析括号内的表达式
                operators.push(token);
            } else if (token.equals(")")) {
                // 遇到右括号，开始构建树
                while (!operators.peek().equals("(")) {
                    processOperator(operands, operators.pop());
                }
                operators.pop();  // 移除 '('
            } else if (token.equals("!") || token.equals("&") || token.equals("|")) {
                // 操作符，处理运算符优先级
                while (!operators.isEmpty() && precedence(operators.peek()) >= precedence(token)) {
                    processOperator(operands, operators.pop());
                }
                operators.push(token);
            } else {
                // 变量，直接入栈
                operands.push(new TreeNode(token));
            }
            index++;
        }

        // 处理剩余的操作符
        while (!operators.isEmpty()) {
            processOperator(operands, operators.pop());
        }

        return operands.isEmpty() ? null : operands.pop();
    }

    // 处理栈顶的运算符
    private static void processOperator(Stack<TreeNode> operands, String operator) {
        TreeNode node = new TreeNode(operator);

        // 对于 '!' 操作符，只有右子树
        if (operator.equals("!")) {
            node.right = operands.pop();
        } else {
            node.right = operands.pop();
            node.left = operands.pop();
        }

        operands.push(node);
    }

    // 运算符的优先级：'!' > '&' > '|'
    private static int precedence(String operator) {
        if (operator.equals("!")) {
            return 3;
        } else if (operator.equals("&")) {
            return 2;
        } else if (operator.equals("|")) {
            return 1;
        }
        return 0;
    }

    // 打印二叉树，用于测试
    public static void printTree(TreeNode node, String indent) {
        if (node != null) {
            System.out.println(indent + node.value);
            printTree(node.left, indent + "  ");
            printTree(node.right, indent + "  ");
        }
    }

    // 测试
    public static void main(String[] args) {
        // 输入是一个已经切分好的token数组
        List<String> tokens = Arrays.asList("x1", "&", "(", "x2", "|", "!", "x3", ")");
        TreeNode root = parse(tokens);
        printTree(root, "");
    }
}

代码解析

  1. parse：接受已经切分好的 token 列表，然后逐步解析这些 token。

    • 使用两个栈：operands 用于存储操作数（布尔变量或子表达式），operators 用于存储操作符。
    • 遇到左括号 ( 时，将其压入 operators 栈中；遇到右括号 ) 时，弹出操作符并处理，直到遇到左括号。
    • 遇到操作符时，比较当前操作符与栈顶操作符的优先级，如果当前操作符优先级更低，则先处理栈中的操作符。
    • 遇到布尔变量时，直接创建一个树节点并压入 operands 栈中。

  2. processOperator：这个方法用于从 operands 栈中弹出两个操作数，并根据操作符创建新的树节点，然后将新节点压入 operands 栈中。如果是 ! 操作符，只处理右子树；如果是 & 或 |，则处理左右子树。

  3. precedence：这是一个简单的优先级比较方法，用于决定操作符的优先级。! 的优先级最高，其次是 &，最后是 |。

  4. printTree：打印二叉树，用于验证构建出来的树结构是否正确。
